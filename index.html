    increment/decrement;
    }

DO-WHILE LOOP


SYNTAX


    initialization ;
    do{
        increment/decrement;
    }
    while(condition) ;



Example


    var i=1;

    do{
        console.log(i);
        i++;
    }
    while(i<=10);
